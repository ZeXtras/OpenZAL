68  public static void copyJar(ZipFile zipFile, File destination, byte[] digest, byte[] signature) throws IOException
69  {
70  ZipOutputStream zipOutputStream = new ZipOutputStream(new FileOutputStream(destination));
71 
72  byte[] buffer = new byte[32*1024];
73  Enumeration<? extends ZipEntry> it = zipFile.entries();
74  while( it.hasMoreElements() )
75  {
76  ZipEntry zipEntry = it.nextElement();
77  InputStream inputStream = zipFile.getInputStream(zipEntry);
78  try
79  {
80  zipOutputStream.putNextEntry(zipEntry);
81  copyStream(buffer, inputStream, zipOutputStream);
82  }
83  finally
84  {
85  inputStream.close();
86  }
87  }

90  zipOutputStream.write(printableByteArray(digest).getBytes("UTF-8"));
91 
92  zipOutputStream.putNextEntry(new ZipEntry(SIGNATURE));
93  zipOutputStream.write(printableByteArray(signature).getBytes("UTF-8"));
94 
95  zipOutputStream.close();
96  }

107  public static String printableByteArray(byte[] buffer)
108  {
109  char[] encoded = new char[2 * buffer.length];
110  for( int i=0; i < buffer.length; ++i )
111  {
112  int value = buffer[i] & 0xFF;

115  }

117  return new String(encoded);

130  public static byte[] computeDigest(ZipFile zipFile) throws NoSuchAlgorithmException, IOException
131  {
132  byte[] buffer = new byte[16*1024];
133  MessageDigest digest = MessageDigest.getInstance("SHA-256");
134  Enumeration<? extends ZipEntry> zipEntries = zipFile.entries();
135  while (zipEntries.hasMoreElements())
136  {
137  ZipEntry entry = zipEntries.nextElement();
138 
139  if (DIGEST.equals(entry.getName()) || SIGNATURE.equals(entry.getName()))
140  {
141  continue;
142  }
143 
144  digest.update(
145  entry.getName().getBytes("UTF-8")
146  );
147  updateDigest(buffer, digest, zipFile.getInputStream(entry));
148  }
149

151  }
