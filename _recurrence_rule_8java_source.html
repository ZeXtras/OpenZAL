127  public List<Integer> getByCalendarDayList(Long startTime, TimeZone timezone)
128  {
129  List<ZRecur.ZWeekDayNum> byDayList = mZRecur.getByDayList();
130  List<Integer> list = new ArrayList<Integer>(byDayList.size());
131 
132  for( ZRecur.ZWeekDayNum weekDayNum : byDayList )
133  {
134  list.add(weekDayNum.mDay.getCalendarDay());
135  }
136 
137  if (list.isEmpty() && Frequency.WEEKLY.equals(getFrequency()) && startTime != null && timezone != null)
138  {
139  Calendar calendar = Calendar.getInstance(timezone);
140  calendar.setTimeInMillis(startTime);
141  list.add(calendar.get(Calendar.DAY_OF_WEEK));
142  }
143 
144  return list;
145  }
