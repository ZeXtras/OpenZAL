64  public static String exceptionToString(Throwable e)
65  {
66  StringBuilder sb = new StringBuilder(128);
67  StackTraceElement elements[] = e.getStackTrace();
68 
69  sb.append(e.toString());
70  sb.append("\n");
71 
72  for (int n = 0; n < elements.length; ++n)
73  {
74  sb.append(" at ");
75  sb.append(elements[n].getClassName());
76  sb.append(".");
77  sb.append(elements[n].getMethodName());
78  sb.append(" (");
79  sb.append(elements[n].getFileName());
80  sb.append(":");
81  sb.append(elements[n].getLineNumber());
82  sb.append(")");
83 
84  if (elements[n].isNativeMethod())
85  {
86  sb.append(" [native]");
87  }
88 
89  sb.append("\n");
90  }
91 
92  Throwable cause = e.getCause();
93  if (cause != null)
94  {
95  sb.append("Caused by: ");
96  sb.append(exceptionToString(cause));
97  }
98 
99  return sb.toString();
100  }

179  public static void deployZimlet(Provisioning provisioning, ZimletFile zimlet) throws IOException
180  {
181  try
182  {
183 /* $if ZimbraVersion < 6.0.8 $
184  ZimletUtil.deployZimletBySoap(zimlet.getZimletPath(), null, null, null, null);
185  $endif$ */
186 
187 /* $if ZimbraVersion >= 6.0.8 && ZimbraVersion <= 8.0.9$
188  ZimletUtil.deployZimletBySoap(zimlet.getZimletPath(), null, null, true);
189  $endif$ */
190 
191 /* $if ZimbraVersion >= 8.5.0 $ */
192  provisioning.flushCache(CacheEntryType.zimlet, null);
193  ZimletUtil.deployZimletLocally(zimlet.toZimbra(com.zimbra.cs.zimlet.ZimletFile.class));
194 
195  FileInputStream in = new FileInputStream(new File(zimlet.getZimletPath()));
196  ZimletFile rawZimlet = new ZimletFile(zimlet.getName(), in);
197 
198  ZimletUtil.ZimletSoapUtil zimletSoapUtil = new ZimletUtil.ZimletSoapUtil();
199  for (Server server : provisioning.getAllServers())
200  {
201  if (!server.toZimbra(com.zimbra.cs.account.Server.class).isLocalServer())
202  {
203  if( !server.hasMailboxService() )
204  {
205  continue;
206  }
207 
208  zimletSoapUtil.deployZimletRemotely(
209  server.toZimbra(com.zimbra.cs.account.Server.class),
210  rawZimlet.getName(),
211  rawZimlet.getZimletContent(),
212  null,
213  true
214  );
215  }
216  }
217  // In Zimbra 8.5 the Zimlet cache may not be consistent. Better flush it again after the deploy.
218  provisioning.flushCache(CacheEntryType.zimlet, null);
219 /* $endif$ */
220  }
221  catch (ServiceException e)
222  {
223  throw ExceptionWrapper.wrap(e);
224  }
225 /* $if ZimbraVersion >= 8.5.0 $ */
226  catch (ZimletException e)
227  {
228  // In Zimbra >= 8.5.0 if something fails, flush the cache.
229  provisioning.flushCache(CacheEntryType.zimlet, null);
230  throw ExceptionWrapper.wrap(e);
231  }
232 /* $endif$ */
233  }
